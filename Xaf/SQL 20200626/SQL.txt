create view vwWarehouseBins
as
(
	select '' + WhsCode as BoKey, WhsCode as BinCode, WhsCode, WhsName, 0 as BinAbsEntry
	, case when Inactive = 'Y' then 0 else 1 end as IsActive
	,'HQ' as CompanyCode 
	from [PUSRAWI_LIVE]..OWHS
)
go
